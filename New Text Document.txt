<script>
        function updateTimeAgo() {
            document.querySelectorAll('.news-card').forEach((card, index) => {
                // مفتاح سحري فريد لكل كارت بناءً على ترتيبه لحفظ وقت النشر التلقائي
                const storageKey = `news_post_time_${index}`;
                let savedTime = localStorage.getItem(storageKey);

                // لو الكارت لسه محطوط حالا ومكتوب فيه كالكوليت، بنسجل وقت اللحظة الحالية فوراً
                const timeElement = card.querySelector('.time-ago');
                if (timeElement && timeElement.innerText === "Calculating..." && !savedTime) {
                    const nowISO = new Date().toISOString();
                    localStorage.setItem(storageKey, nowISO);
                    card.setAttribute('data-time', nowISO);
                    savedTime = nowISO;
                }

                // قراءة تاريخ الخبر (سواء الأصلي أو المحفوظ تلقائياً)
                const timeAttr = savedTime || card.getAttribute('data-time');
                if (!timeAttr) return;

                const postDate = new Date(timeAttr);
                const nowDate = new Date();
                const diffMs = nowDate - postDate;
                
                // الحسبة الدقيقة بالثواني والدقائق والساعات
                const diffSecs = Math.floor(diffMs / 1000);
                const diffMins = Math.floor(diffSecs / 60);
                const diffHours = Math.floor(diffMins / 60);
                const diffDays = Math.floor(diffHours / 24);

                let timeString = "";
                
                // العداد هيبدأ يعد من أول كام ثانية فوراً!
                if (diffSecs < 60) { 
                    timeString = diffSecs <= 3 ? "Just Now" : `${diffSecs} seconds ago`; 
                }
                else if (diffMins < 60) { 
                    timeString = diffMins === 1 ? "1 minute ago" : `${diffMins} minutes ago`; 
                }
                else if (diffHours < 24) { 
                    timeString = diffHours === 1 ? "1 hour ago" : `${diffHours} hours ago`; 
                }
                else { 
                    timeString = diffDays === 1 ? "Yesterday" : `${diffDays} days ago`; 
                }

                if (timeElement) { timeElement.innerText = timeString; }
            });
        }
        
        // تشغيل العداد فوراً عند تحميل الصفحة
        updateTimeAgo();
        
        // خليناه يحدث كل ثانية واحدة (1000ms) بدل دقيقة عشان تشوف الثواني وهي بتعد قدام عينك!
        setInterval(updateTimeAgo, 1000); 
    </script>
